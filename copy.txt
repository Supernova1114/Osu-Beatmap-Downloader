public static String fileReader()
    {

        ArrayList<String> settingsArray = new ArrayList<String>();


        // The name of the file to open.
        String fileName = settDir;

        // This will reference one line at a time
        String line = null;

        try {
            // FileReader reads text files in the default encoding.
            FileReader fileReader =
                    new FileReader(fileName);

            // Always wrap FileReader in BufferedReader.
            BufferedReader bufferedReader =
                    new BufferedReader(fileReader);

            while((line = bufferedReader.readLine()) != null) {
                settingsArray.add(line);
            }

            // Always close files.
            bufferedReader.close();

            String settingName = settingsArray.get(0);
            System.out.println(settingName);
            String output = settingName.substring( settingName.indexOf("\"")+1, settingName.length()-1 );

            if (output.substring(output.length()-1).equals("\\")){
                return output; //downloadPath
            }
            else {
                return output + "\\";
            }
        }
        catch(FileNotFoundException ex) {
            System.out.println(
                    "Unable to open file '" +
                            fileName + "'");
            JOptionPane.showMessageDialog(null,"Error: Cannot Find Settings Folder!\nFolder Creation Not Implemented In Version 1!");
            Main.scraper.driver.quit();
            System.exit(1);
        }
        catch(IOException ex) {
            System.out.println(
                    "Error reading file '"
                            + fileName + "'");
            // Or we could just do this:
            // ex.printStackTrace();
        }
        return "";
    }//fileReader()
